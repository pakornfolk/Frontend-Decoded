JavaScript Essentials

Code Example var vs let vs const — ความแตกต่างที่สำคัญ

VAR — หลีกเลี่ยง

Function-scoped, hoisted, reassignable — พฤติกรรมแปลกและ bug ได้ง่าย

LET — ใช้เมื่อต้องเปลี่ยนค่า

Block-scoped, ไม่ถูก hoisted, reassignable — ใช้กับ counter, loop

CONST — default choice

Block-scoped, ไม่ reassign — ใช้เป็น default เสมอ เว้นแต่รู้ว่าต้องเปลี่ยน

variables.js 
// ❌ var — hoisting ทำให้งงได้ง่าย
console.log(name); // undefined (ไม่ error!)
var name = "Alice";

// ✓ let — ใช้เมื่อต้องเปลี่ยนค่า
let count = 0;
count++; // OK

for (let i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 100);
  // ✓ พิมพ์ 0, 1, 2 ถูกต้อง
}

// ✓ const — ใช้เป็น default
const user = { name: "Bob", age: 25 };
user.age = 26; // OK — object ยังแก้ได้
// user = {} ❌ ไม่ได้ — reassign ไม่ได้

// Destructuring
const { name: userName, age } = user;
const [first, ...rest] = [1, 2, 3, 4];

async-await.js 
// Fetch API + Async/Await
async function getUser(id) {
  try {
    const res = await fetch(
      `https://api.example.com/users/${id}`
    );

    // ตรวจสอบ status ก่อนเสมอ
    if (!res.ok) {
      throw new Error(`HTTP error: ${res.status}`);
    }

    const data = await res.json();
    return data;

  } catch (err) {
    console.error('Fetch failed:', err);
    throw err; // re-throw เพื่อให้ caller จัดการ
  }
}

// Parallel fetching
const [users, posts] = await Promise.all([
  fetch('/api/users').then(r => r.json()),
  fetch('/api/posts').then(r => r.json()),
]);
Code Example Array Methods — ใช้บ่อยที่สุด
array-methods.js 
const products = [
  { name: 'Phone',  price: 15000, inStock: true  },
  { name: 'Tablet', price: 25000, inStock: false },
  { name: 'Watch',  price: 8000,  inStock: true  },
];

// map — แปลงทุก element
const names = products
  .map(p => p.name);
// ['Phone', 'Tablet', 'Watch']

// filter — กรอง
const available = products
  .filter(p => p.inStock);
// [{Phone}, {Watch}]

// find — หาตัวแรกที่ match
const cheap = products
  .find(p => p.price < 10000);
// {Watch}// reduce — สรุปเป็นค่าเดียว
const total = products
  .reduce((sum, p) => sum + p.price, 0);
// 48000

// some / every
const hasStock = products.some(p => p.inStock);
// true
const allStock = products.every(p => p.inStock);
// false

// Chain methods
const result = products
  .filter(p => p.inStock)
  .map(p => ({
    ...p,
    priceFormatted: `฿${p.price.toLocaleString()}`
  }))
  .sort((a, b) => a.price - b.price);

/* ── JS: Debounce — ป้องกัน over-firing ─────── */
function debounce(fn, delay) {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
}

const onSearch = debounce((query) => {
  fetchResults(query);
}, 300); // รอ 300ms หลังพิมพ์หยุด

/* ── JS: requestAnimationFrame ────────────────── */
// animation ให้ใช้ rAF แทน setInterval

  // update animation

  requestAnimationFrame(animate);
}
